\chapter*{Danh mục ký hiệu và từ viết tắt}
\addcontentsline{toc}{chapter}{Danh mục ký hiệu và từ viết tắt}

\begin{itemize}[leftmargin=3cm, labelsep=1cm]
    \item[\textbf{API}] Application Programming Interface (Giao diện lập trình ứng dụng)
    \item[\textbf{AWS}] Amazon Web Services
    \item[\textbf{BI}] Business Intelligence (Trí tuệ doanh nghiệp)
    \item[\textbf{CI/CD}] Continuous Integration / Continuous Deployment
    \item[\textbf{CSDL}] Cơ sở dữ liệu
    \item[\textbf{ELK}] Elasticsearch, Logstash, Kibana
    \item[\textbf{ESB}] Enterprise Service Bus (Trục dịch vụ doanh nghiệp)
    \item[\textbf{HRM}] Human Resource Management (Hệ thống quản lý nhân sự)
    \item[\textbf{JWT}] JSON Web Token
    \item[\textbf{LMS}] Learning Management System (Hệ thống quản lý học tập)
    \item[\textbf{MVP}] Minimum Viable Product (Sản phẩm khả thi tối thiểu)
    \item[\textbf{ODS}] Operational Data Store (Kho dữ liệu hoạt động)
    \item[\textbf{OLAP}] Online Analytical Processing (Xử lý phân tích trực tuyến)
    \item[\textbf{OLTP}] Online Transaction Processing (Xử lý giao dịch trực tuyến)
    \item[\textbf{P2P}] Point-to-Point (Mô hình điểm - điểm)
    \item[\textbf{RBAC}] Role-Based Access Control (Kiểm soát truy cập dựa trên vai trò)
    \item[\textbf{SIS}] Student Information System (Hệ thống thông tin sinh viên)
\end{itemize}
